#build the container
docker build -t py-lab-img .

#run the container
docker run -P --rm --name docker-lab-1 py-lab-img:latest

#take the ip and port from the container 'docker ps -a'

#enter into the container
docker exec -it <container-name> bash

#enter into the supervisor console
supervisorctl

#restart the process inside the supervisor console
restart python

#restart the process outside the supervisor console
supervisorctl restart python
